# 回款台账(ledger) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 点亮"回款台账"页（`/ledger`）——项目级台账：汇总条 + 状态行 + 分层卡 + 搜索/区间/状态筛选 + 11 列项目表（行点击展开下钻回款节点明细），并复用 B9 的 CF 列头筛选。

**Architecture:** 计算逻辑放纯函数 `lib/ledger.ts`（数据源/过滤/汇总/分层/状态计数）。表格用自有 `<table>`（贴近旧 DOM、便于挂 CF 列头与行展开，沿用 B9 PlanBoard 的自定义表格做法）拆为 `LedgerTable.vue`。`LedgerView.vue` owns 筛选控件 + 三组指标条 + 装配。CF 复用 B9 的 `stores/crossFilter` + `components/ColumnFilter.vue`（单表 `ledgerTable`，无联动组）。

**Tech Stack:** Vue 3 `<script setup>` + TS + Pinia + Element Plus (el-input/el-select/el-button) + Vitest。

**忠实移植基准（旧 app.js）：** `initLedger`(3957) / `filterLedger`(4293) / `toggleLedgerRow`(4155) / `renderLedgerDrilldown`(4207) / `_filteredRawNodes`(155) / `CF.register('ledgerTable')`(4390)。

**关键忠实性要点（务必照搬，避免想当然）：**
- 台账数据源是 `_filteredRawNodes()` = **仅纳管过滤**（不含年份/视角）。`initLedger` 末尾调用 `filterLedger()`，后者用 `groupByProject(_filteredRawNodes())` 重算并覆盖一切，所以渲染态恒为"纳管过滤"。→ 本期 ledger 不接 filterStore.filteredNodes（那含年份/视角），而用新纯函数 `naguanFilter`。
- 三组指标条（汇总/分层卡/状态行）均基于"经搜索/区间/状态/CF 过滤后的 projs"重算（`filterLedger` 4314-4355），不是固定值。
- 列表按 `projectAmount` 降序、`slice(0,500)`。
- 分层卡按 `p.tier` 分组；而"区间"下拉过滤按 `p.nodes.some(n=>n.tier===tf)`（两者口径不同，照搬）。
- 行展开下钻：每次只展开一行；展示该项目的"回款节点明细"（节点/计划日期/待回款(元)/实际比例/状态）。

**展示从简（已记录，非偏差）：**
- 下钻里旧版的"项目全字段横向行 + 设置展示字段(列可见性)UI"延后 B-opt；本期下钻只渲染"回款节点明细"（核心钻取内容）。
- tier/status 徽章配色延后 B-opt（与既有 tab 一致，纯文本展示）。
- 导出 Excel 延后 B-opt。
- 旧 CF 在 `remainAmount` 列用了不存在的 key（项目对象实为 `remainingAmount`），本期用正确 key 接 CF（修正而非照搬该笔误）。

import type { RawNode } from '@/types/analysis'
import type { ProjectAgg } from './dashboardStats'

/** 仅纳管过滤（忠实移植 _filteredRawNodes：台账数据源不含年份/视角过滤）。 */
export function naguanFilter(
  rawNodes: RawNode[],
  naguanOn: boolean,
  naguanExclude: Record<string, boolean>,
): RawNode[] {
  if (!naguanOn || !naguanExclude) return rawNodes
  return rawNodes.filter((n) => !naguanExclude[(n as Record<string, any>).projectId])
}

export interface LedgerFilterOpts {
  search: string
  tier: string
  status: string
}

/** 忠实移植 filterLedger 的搜索/区间/状态过滤 + 按项目金额降序。CF 列筛选在组件层另用 applyColumnFilters。 */
export function filterLedgerProjects(projs: ProjectAgg[], opts: LedgerFilterOpts): ProjectAgg[] {
  const q = (opts.search || '').toLowerCase()
  let out = projs
  if (opts.tier)
    out = out.filter(
      (p) => p.nodes && p.nodes.some((n) => (n as Record<string, any>).tier === opts.tier),
    )
  if (opts.status) out = out.filter((p) => p.paymentStatus === opts.status)
  if (q)
    out = out.filter((p) =>
      (String(p.projectId) + p.projectName + p.projectManager + p.orgL4).toLowerCase().includes(q),
    )
  return [...out].sort((a, b) => (b.projectAmount || 0) - (a.projectAmount || 0))
}

export interface LedgerSummary {
  projectCount: number
  totalExp: number
  totalAct: number
  totalRem: number
  rate: number
}
export function ledgerSummary(projs: ProjectAgg[]): LedgerSummary {
  const totalExp = projs.reduce((s, p) => s + (p.expectedPayment || 0), 0)
  const totalAct = projs.reduce((s, p) => s + (p.actualPayment || 0), 0)
  return {
    projectCount: projs.length,
    totalExp,
    totalAct,
    totalRem: totalExp - totalAct,
    rate: totalExp > 0 ? totalAct / totalExp : 0,
  }
}

const LEDGER_TIERS = ['100万以上', '50-100万', '50万以下']

export interface LedgerTierStat {
  tier: string
  count: number
  expWan: number
  remWan: number
}
/** 忠实移植 filterLedger 的分层卡：按 p.tier 分组（项目级 tier，单位万）。 */
export function ledgerTierStats(projs: ProjectAgg[]): LedgerTierStat[] {
  return LEDGER_TIERS.map((t) => {
    const tp = projs.filter((p) => p.tier === t)
    const exp = tp.reduce((s, p) => s + (p.expectedPayment || 0), 0)
    const act = tp.reduce((s, p) => s + (p.actualPayment || 0), 0)
    return { tier: t, count: tp.length, expWan: exp / 10000, remWan: (exp - act) / 10000 }
  })
}

export interface LedgerStatusCounts {
  canAdvance: number
  reachedCondition: number
  advance: number
  fullPaid: number
  delayed: number
  onTime: number
}
/** 项目级状态计数（按 paymentStatus）。忠实移植 ledger-status-row 联动计数。 */
export function ledgerStatusCounts(projs: ProjectAgg[]): LedgerStatusCounts {
  const c = (s: string) => projs.filter((p) => p.paymentStatus === s).length
  return {
    canAdvance: c('加资源可提前'),
    reachedCondition: c('达到回款条件'),
    advance: c('已提前回款'),
    fullPaid: c('已全额回款'),
    delayed: c('延期'),
    onTime: c('正常实施中'),
  }
}

interface LedgerCol {
  key: string
  label: string
  formatter?: (value: any, row: Record<string, any>) => string
}

const props = defineProps<{
  tableId: string
  projects: Record<string, any>[]
  columns: LedgerCol[]
  sourceRows: Record<string, any>[]

      (data.data?.naguanExclude ?? {}) as Record<string, boolean>,
    ),

    tier: tierSel.value,
    status: statusSel.value,

)

const tierStats = computed(() => ledgerTierStats(displayed.value as any))
const statusCounts = computed(() => ledgerStatusCounts(displayed.value as any))

    formatter: (_v: any, row: Record<string, any>) =>
      pct(row.expectedPayment > 0 ? row.actualPayment / row.expectedPayment : 0),

  { key: 'paymentStatus', label: '状态' },

    <LedgerTable

      :projects="displayed"
